git init -- инициализация работы с гитом, начало проекта
git remote add origin {url} -- привязка проекта к удаленному репозиторию
git checkout -b <название ветки> -- создать новую ветку и переключиться на нее 
git commit --amend -- изменение текста последнего коммита 
git status -- посмотреть статус и название ветки 
git switch -с <название ветки> -- переключиться на ветку
git branch --list --all  -- посмотреть какие есть ветки 
git rm --cached <файл> -- удалить файл из коммита
git push origin --delete <название ветки> -- удалить ветку в гите
